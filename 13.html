Arrow Function

////////// 箭頭函式與return的用法 //////////

let myArr = [1, 2, 3, 4, 5,];
let mapArr = myArr.map(function (value, key) {
return value * 2;
})


console.log('mapArr', mapArr);

let mapArrowArr = myArr.map(value => value * 2);

//參數只有一個 ( ) 省略
//( value , key ) => { return value * 2 }

console.log('mapArrowArr', mapArrowArr);
